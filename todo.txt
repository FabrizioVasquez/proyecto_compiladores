        programa : lista_declaracion;
        lista_declaracion : lista_declaracion declaracion;
        declaracion : var_declaracion | fun_declaracion;
        var_declaracion : entero IDENTIFICADOR PUNTOYCOMA | entero IDENTIFICADOR [NUMERO] PUNTOYCOMA ;
        tipo : entero | sin_tipo;
        fun_declaracion : tipo IDENTIFICADOR ABREPAR params CIERRAPAR sent_compuesta;
        params : lista_params | sin_tipo;
        lista_params : lista_params COMA param | param;
        param : tipo IDENTIFICADOR | tipo IDENTIFICADOR ABRECOR CIERRACOR;
    sent_compuesta : ABRELLAV declaracion_local lista_sentencias CIERRALLAV;
    declaracion_local : declaracion_local var_declaracion | vacio;
    lista_sentencias : lista_sentencias sentencia | vacio;
    sentencia : sentencia_expresion | sentencia_seleccion | sentencia_iteracion | sentencia_retorno;
    sentencia_expresion : expresion PUNTOYCOMA | PUNTOYCOMA;
    sentencia_seleccion : SI ABREPAR expresion CIERRAPAR sentencia | SI ABREPAR extension CIERRAPAR sentencia SINO sentencia;
    sentencia_iteracion : MIENTRAS ABREPAR expresion CIERRAPAR ABRELLAV lista_sentencias CIERRALLAV;
    sentencia_retorno : retorno PUNTOYCOMA | retorno expresion PUNTOYCOMA;
expresion : var IGUAL expresion | expresion_simple;
var : IDENTIFICADOR | IDENTIFICADOR ABRECOR expresion CIERRACOR;
expresion_simple : expresion_aditiva RELOP expresion_aditiva | expresion_aditiva;
expresion_aditiva : expresion_aditiva ADDOP term | term;
term : term MULOP factor | factor;
factor : ABREPAR expresion CIERRAPAR | var | call | NUMERO;
call : IDENTIFICADOR ABREPAR  args CIERRAPAR;
args : lista_arg | vacio;
lista_arg : lista_arg COMA expresion | expresion;
MULOP : MUL | DIV;
ADDOP : SUM | RES;
